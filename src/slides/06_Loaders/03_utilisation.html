Utilisation des loaders

Installation

npm install css-loader --save-dev

utilisation dans un import

var style = require('css!less!./style.less');

utilisation dans la configuration

// webpack.config.js
loaders: [
    {
        test: /\.less$/,
        loader: 'css!less'
    }
]
// index.js
var style = require('./style.less');
